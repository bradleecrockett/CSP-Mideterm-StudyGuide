# init a counter

    if i % 2 == 1:
        count += 1

# Accumulate a string
# init an empty str
vowels = ""
letter = ""
while not(letter == 'quit'):
    letter = input("Enter a letter, Enter 'quit' to quit: ")
    if letter in 'aeiou':
        print("That's a vowel!")
        vowels += letter
    else:
        print("Not a vowel")
print("Vowels entered were:", vowels)
